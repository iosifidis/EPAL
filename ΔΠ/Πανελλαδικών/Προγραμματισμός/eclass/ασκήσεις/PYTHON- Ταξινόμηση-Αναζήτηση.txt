def search(L,key):
    first=0
    last=len(L)-1
    pos=-1
    while pos=-1 and first<=last:
        mid=(first+last)/2
        if L[mid]==key:
            pos=mid
        elif L[mid]>key:
            last=mid-1
        else:
            first=mid+1
    return pos
